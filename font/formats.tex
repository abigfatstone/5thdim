\def\@part[#1]#2{%
    \ifnum \c@secnumdepth >-2\relax
      \refstepcounter{part}%
      \addcontentsline{toc}{part}{\thepart\hspace{1em}#1}%
    \else
      \addcontentsline{toc}{part}{#1}%
    \fi
    \markboth{}{}%
    \vspace*{3.5cm}
    {\centering
     \interlinepenalty \@M
     \normalfont
     \ifnum \c@secnumdepth >-2\relax
       \Huge\bfseries\color{black}\thepart\par
       \vspace{1.5em}
       \Huge\bfseries\itshape\color{black}#2%
     \else
       \Huge\bfseries\itshape\color{black} #2%
     \fi
     \par}%
    \ifx\@partguide\@empty\else
      \vspace{3em}
      {\centering
       \parbox{0.78\textwidth}{%
         \setstretch{1.18}\small\color{darkGray}\raggedright
         \textbf{\ifdefined\ENGLISH Guide\else 导读\fi}\par
         \vspace{0.7em}
         \@partguide
       }\par}%
    \fi
    \global\let\@partguide\@empty
    \vspace{2.2cm}
    \@endpart}

\def\@spart#1{%
    \vspace*{4.2cm}
    {\centering
     \interlinepenalty \@M
     \normalfont
     \Huge\bfseries\itshape\color{black} #1%
     \par}%
    \global\let\@partguide\@empty
    \vspace{2.2cm}
    \@endpart}

% Chapter 标题重定义
\renewcommand\chapter{%
  \if@openright\cleardoublepage\else\clearpage\fi
  \global\@topnum\z@
  \@afterindentfalse
  \secdef\@chapter\@schapter}

\def\@chapter[#1]#2{%
  \ifnum \c@secnumdepth >\m@ne
    \refstepcounter{chapter}%
    \typeout{\@chapapp\space\thechapter.}%
    \addcontentsline{toc}{chapter}%
      {\protect\numberline{\ifdefined\ENGLISH Chapter~\thechapter\else 第\zhnumber{\thechapter}章\fi}#1}%
  \else
    \addcontentsline{toc}{chapter}{#1}%
  \fi
  \chaptermark{#1}%
  \thispagestyle{plain}%
  \@makechapterhead{#2}%
  \@afterheading}

\def\@makechapterhead#1{%
  \vspace*{30\p@}%
  {\parindent \z@ \raggedright \normalfont
    \ifnum \c@secnumdepth >\m@ne
      \LARGE\bfseries\color{darkGray} \ifdefined\ENGLISH Chapter~\thechapter\else 第\zhnumber{\thechapter}章\fi\quad
    \fi
    \LARGE\bfseries\color{darkGray} #1\par\nobreak
    \vskip 20\p@
  }}

\def\@schapter#1{%
  \thispagestyle{plain}%
  \global\@topnum\z@
  \@afterindentfalse
  \@makeschapterhead{#1}%
  \@afterheading}

\def\@makeschapterhead#1{%
  \vspace*{30\p@}%
  {\parindent \z@ \raggedright
    \normalfont
    \interlinepenalty\@M
    \LARGE \bfseries\color{darkGray} #1\par\nobreak
    \vskip 20\p@
  }}
